```go
func main() {                                                                                                                           
                                                                                                                                        
  x := []int{1, 2, 3}                                                                                                                   
                                                                                  
  for _, element := range x {                                                                                                           
    fmt.Print(element)                                                                                                                  
                                                                                                                                        
  }                                                                                                                                     
}
```
